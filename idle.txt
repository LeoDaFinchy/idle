Let's make a game!
  name:My Totally Original Game
  
Resources
  *charm
    name: Charm
    
  *cash
    name: Cash
  
  *brine
    name: Brine
    
  *brineValue
    name: Brine Value
    startWith: 0.1
  
Buttons
  *charmButton
    name: Charm!
    on click:gain 1 charm
    
  *sellBrineButton
    name: Sell Brine
    req:(brine >= 1)
    on click:gain (brine * brineValue) cash
    on click:lose (brine) brine

Buildings
  *visitor
    cost:10 charm
    passive:gain 0.1 brine
    
  *member
    cost:100 charm,1 visitor
    on tick:gain 1 brine

  *regular
    cost:1000 charm,1 member
    on tick:gain 10 brine
    
  *staff
    cost:100 cash
    on tick:
      if (cash >= 1)
        gain 1 charm
        lose 1 cash
      end
    end
  
  *brineSeller
    cost: 100 cash
    on tick:
      if (brine > 1)
        gain (brineValue * 0.9) cash
        lose 1 brine
      end
    end
    
Upgrades
  *towels
    cost:100 cash
    passive:multiply yield of brine by 2
  *heaters
    cost:1000 cash
    passive:multiply yield of brine by 2
